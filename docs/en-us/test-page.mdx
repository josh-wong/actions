### Python example

```python
# Sample Python function for testing
def greet_user(name, language="en"):
    """
    Greets the user in the specified language.

    Args:
        name (str): The user's name
        language (str): Language code (default: "en")

    Returns:
        str: Formatted greeting message
    """
    greetings = {
        "en": f"Hello, {name}!",
        "ja": f"こんにちは、{name}さん！",
        "es": f"¡Hola, {name}!"
    }

    return greetings.get(language, greetings["en"])

# Usage example
print(greet_user("Alice", "ja"))
```

### JavaScript example

```javascript
// Configuration object for API endpoints
const apiConfig = {
    baseUrl: 'https://api.example.com',
    version: 'v1',
    timeout: 5000,

    // Method to build full endpoint URL
    buildUrl(endpoint) {
        return `${this.baseUrl}/${this.version}/${endpoint}`;
    }
};

// Async function to fetch user data
async function fetchUserData(userId) {
    try {
        const response = await fetch(apiConfig.buildUrl(`users/${userId}`));
        const userData = await response.json();
        return userData;
    } catch (error) {
        console.error('Failed to fetch user data:', error);
        throw error;
    }
}
```
